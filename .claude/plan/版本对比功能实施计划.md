#### 1. 输入校验（DiffController）

```java
@GetMapping("/compare")
public Result<List<DiffUtil.DiffResult>> compareVersions(
        @RequestParam Long version1Id,
        @RequestParam Long version2Id) {

    // 1. 参数校验
    if (version1Id == null || version2Id == null) {
        throw new BadRequestException("版本ID不能为空");
    }
    if (version1Id.equals(version2Id)) {
        throw new BadRequestException("不能对比同一版本");
    }

    // 2. 获取当前用户
    Long currentUserId = SecurityUtil.getCurrentUserId();
    String role = SecurityUtil.getCurrentUserRole();

    // 3. 调用服务层（包含权限校验）
    List<DiffUtil.DiffResult> diff = thesisService.compareVersionsWithAuth(
            version1Id, version2Id, currentUserId, role);

    return Result.success(diff);
}
```

#### 2. 权限校验逻辑（ThesisService）

```java
public List<DiffUtil.DiffResult> compareVersionsWithAuth(
        Long v1Id, Long v2Id, Long userId, String role) {

    // 1. 加载版本
    ThesisVersion v1 = versionMapper.selectById(v1Id);
    ThesisVersion v2 = versionMapper.selectById(v2Id);
    if (v1 == null || v2 == null) {
        throw new NotFoundException("版本不存在");
    }

    // 2. 跨论文校验
    if (!v1.getThesisId().equals(v2.getThesisId())) {
        throw new BadRequestException("不能对比不同论文的版本");
    }

    // 3. 加载论文
    Thesis thesis = thesisMapper.selectById(v1.getThesisId());
    if (thesis == null) {
        throw new NotFoundException("论文不存在");
    }

    // 4. 学生权限校验
    if ("STUDENT".equals(role) && !thesis.getStudentId().equals(userId)) {
        throw new ForbiddenException("无权访问他人论文");
    }

    // 5. contentHash 短路
    if (v1.getContentHash() != null &&
        v1.getContentHash().equals(v2.getContentHash())) {
        return Collections.emptyList(); // 无差异
    }

    // 6. 执行对比
    File f1 = getVersionFile(v1Id);
    File f2 = getVersionFile(v2Id);
    return diffUtil.compareFiles(f1.getAbsolutePath(), f2.getAbsolutePath());
}
```

#### 4. VersionComparer.vue 组件

**组件结构**：
- 三个 Tab：统计摘要 / 并排对比 / 合并视图
- 响应式：移动端隐藏并排视图
- 懒加载：使用 defineAsyncComponent

**关键逻辑**：

```vue
<template>
  <div class="version-comparer">
    <el-skeleton v-if="isLoading" :rows="10" />
    <el-alert v-else-if="error" :title="error" type="error" />

    <el-tabs v-else v-model="activeTab">
      <!-- 统计摘要 -->
      <el-tab-pane label="统计摘要" name="summary">
        <el-row :gutter="20">
          <el-col :span="8">
            <el-statistic title="新增" :value="summary.added" />
          </el-col>
          <el-col :span="8">
            <el-statistic title="删除" :value="summary.deleted" />
          </el-col>
          <el-col :span="8">
            <el-statistic title="修改" :value="summary.modified" />
          </el-col>
        </el-row>
      </el-tab-pane>

      <!-- 并排对比（桌面端） -->
      <el-tab-pane v-if="!isMobile" label="并排对比" name="side-by-side">
        <div v-html="sideBySideHtml"></div>
      </el-tab-pane>

      <!-- 合并视图 -->
      <el-tab-pane label="合并视图" name="unified">
        <div v-html="unifiedHtml"></div>
      </el-tab-pane>
    </el-tabs>
  </div>
</template>

<script setup>
import { ref, onMounted, computed } from 'vue'
import { getDiff } from '@/api/thesis'
import { Diff2Html } from 'diff2html'
import 'diff2html/bundles/css/diff2html.min.css'

const props = defineProps({
  version1Id: String,
  version2Id: String
})

const isLoading = ref(true)
const error = ref(null)
const backendData = ref([])
const isMobile = ref(window.innerWidth < 768)
const activeTab = ref('summary')
const sideBySideHtml = ref('')
const unifiedHtml = ref('')

// 统计摘要计算
const summary = computed(() => {
  let added = 0, deleted = 0, modified = 0
  backendData.value.forEach(chunk => {
    if (chunk.type === 'INSERT') added += chunk.revisedLines.length
    else if (chunk.type === 'DELETE') deleted += chunk.originalLines.length
    else if (chunk.type === 'CHANGE') modified += chunk.revisedLines.length
  })
  return { added, deleted, modified }
})

// 数据获取
async function fetchData() {
  try {
    const res = await getDiff(props.version1Id, props.version2Id)
    backendData.value = res.data
    renderDiffs()
  } catch (e) {
    error.value = '数据加载失败：' + (e.response?.data?.message || e.message)
  } finally {
    isLoading.value = false
  }
}

// 渲染 Diff 视图
function renderDiffs() {
  const diffJson = convertBackendToDiff2Html(backendData.value)

  if (!isMobile.value) {
    sideBySideHtml.value = Diff2Html.html(diffJson, {
      drawFileList: false,
      outputFormat: 'side-by-side',
      matching: 'lines'
    })
  }

  unifiedHtml.value = Diff2Html.html(diffJson, {
    drawFileList: false,
    outputFormat: 'line-by-line'
  })
}

// 数据适配器：后端格式 → diff2html 格式
function convertBackendToDiff2Html(data) {
  if (!data || data.length === 0) return []

  const lines = []
  data.forEach(chunk => {
    let oldNum = chunk.originalPosition
    let newNum = chunk.revisedPosition

    if (chunk.type === 'DELETE') {
      chunk.originalLines.forEach(line => {
        lines.push({
          type: 'delete',
          content: '-' + line,
          oldNumber: oldNum++,
          newNumber: null
        })
      })
    } else if (chunk.type === 'INSERT') {
      chunk.revisedLines.forEach(line => {
        lines.push({
          type: 'insert',
          content: '+' + line,
          oldNumber: null,
          newNumber: newNum++
        })
      })
    } else if (chunk.type === 'CHANGE') {
      chunk.originalLines.forEach(line => {
        lines.push({
          type: 'delete',
          content: '-' + line,
          oldNumber: oldNum++,
          newNumber: null
        })
      })
      chunk.revisedLines.forEach(line => {
        lines.push({
          type: 'insert',
          content: '+' + line,
          oldNumber: null,
          newNumber: newNum++
        })
      })
    }
  })

  return [{
    blocks: [{
      header: `@@ -1,${lines.length} +1,${lines.length} @@`,
      lines: lines
    }],
    oldName: '旧版本',
    newName: '新版本',
    language: 'plaintext'
  }]
}

onMounted(() => {
  fetchData()
  activeTab.value = isMobile.value ? 'unified' : 'summary'
})

// 响应式监听
window.addEventListener('resize', () => {
  isMobile.value = window.innerWidth < 768
})
</script>

<style scoped>
.d2h-file-header { display: none; }
</style>
```

| ≥ 768px | 显示所有Tab，默认"统计摘要" |
